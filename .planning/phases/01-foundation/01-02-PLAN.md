       Environment string
   }

   func Load() *Config {
       return &Config{

           HTTPPort:    getEnvInt("HTTP_PORT", 8080),
           GRPCPort:    getEnvInt("GRPC_PORT", 50051),
           Environment: getEnv("GO_ENV", "development"),
       }
   }

   func getEnv(key, fallback string) string {
       if v := os.Getenv(key); v != "" {
           return v
       }
       return fallback
   }

   func getEnvInt(key string, fallback int) int {
       if v := os.Getenv(key); v != "" {
           if i, err := strconv.Atoi(v); err == nil {
               return i
           }
       }
       return fallback
   }

5. Create `internal/db/pool.go`:
   ```go
   package db

   import (
       "context"
       "fmt"

       "github.com/jackc/pgx/v5/pgxpool"
   )

   func NewPool(ctx context.Context, databaseURL string) (*pgxpool.Pool, error) {
       config, err := pgxpool.ParseConfig(databaseURL)
       if err != nil {
           return nil, fmt.Errorf("parse database config: %w", err)
       }

       config.MaxConns = 10
       config.MinConns = 2

       pool, err := pgxpool.NewWithConfig(ctx, config)
       if err != nil {
           return nil, fmt.Errorf("create connection pool: %w", err)
       }

       if err := pool.Ping(ctx); err != nil {
           pool.Close()
           return nil, fmt.Errorf("ping database: %w", err)
       }

       return pool, nil
   }

   func HealthCheck(ctx context.Context, pool *pgxpool.Pool) error {
       return pool.Ping(ctx)
   }
   ```

6. Create `internal/http/health.go`:
   ```go
   package http

   import (
       "fmt"
       "log"
       "net/http"

       "github.com/gin-gonic/gin"
       "github.com/jackc/pgx/v5/pgxpool"
   )

   func NewRouter(pool *pgxpool.Pool) *gin.Engine {
       gin.SetMode(gin.ReleaseMode)
       r := gin.New()
       r.Use(gin.Recovery())

       r.GET("/health", healthHandler(pool))

       return r
   }

   func healthHandler(pool *pgxpool.Pool) gin.HandlerFunc {
       return func(c *gin.Context) {
           status := "ok"
           httpStatus := http.StatusOK

           if pool != nil {
               if err := pool.Ping(c.Request.Context()); err != nil {
                   status = "degraded"
                   httpStatus = http.StatusServiceUnavailable
               }
           }

           c.JSON(httpStatus, gin.H{"status": status})
       }
   }

   func StartHealthServer(port int, pool *pgxpool.Pool) {
       router := NewRouter(pool)
       addr := fmt.Sprintf(":%d", port)
       log.Printf("health server listening on %s", addr)
       if err := router.Run(addr); err != nil {
           log.Fatalf("health server failed: %v", err)
       }
   }
   ```
   IMPORTANT: The health endpoint returns ONLY `{"status":"ok"}` or `{"status":"degraded"}`. No version info, no environment name, no internal details.

1. Create `cmd/worker/main.go`:
   ```go
   package main

   import (
       "context"
       "log"
       "os"
       "os/signal"
       "syscall"

       "github.com/sovra/worker/internal/config"
       "github.com/sovra/worker/internal/db"
       workergrpc "github.com/sovra/worker/internal/grpc"
       workerhttp "github.com/sovra/worker/internal/http"
   )

   func main() {
       cfg := config.Load()

       log.Printf("starting sovra worker (env=%s)", cfg.Environment)

       // Connect to database
       // SECURITY: Never log the database URL - only connection status
       pool, err := db.NewPool(context.Background(), cfg.DatabaseURL)
       if err != nil {
           log.Printf("warning: database connection failed: %v", err)
           log.Println("worker starting without database - health checks will report degraded")
           // Start without DB - allows container health checks to work during startup
           pool = nil
       } else {
           log.Println("database connected")
           defer pool.Close()
       }

       // Start HTTP health server
       go workerhttp.StartHealthServer(cfg.HTTPPort, pool)

       // Start gRPC server
       go workergrpc.StartServer(cfg.GRPCPort, pool)

       // Graceful shutdown
       quit := make(chan os.Signal, 1)
       signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
       sig := <-quit
       log.Printf("received signal %s, shutting down", sig)
   }
   ```
   NOTE: The worker starts even if DB is unavailable (returns degraded health). This allows Docker health checks to work during Supabase startup. DB failure is logged as warning, not fatal.
