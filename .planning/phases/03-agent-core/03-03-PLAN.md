    1. Create `packages/web/app/api/chat/route.ts`:
       ```typescript
       import { streamText } from 'ai'
       import { createSupabaseServerClient } from '@/lib/auth/server'
       import { getProvider, initProviders } from '@/lib/ai/registry'

       export async function POST(req: Request) {
         const supabase = await createSupabaseServerClient()
         const { data: { user } } = await supabase.auth.getUser()
         if (!user) return new Response('Unauthorized', { status: 401 })

         const { messages, agentId, conversationId } = await req.json()

         // Validate required fields
         if (!agentId || !messages?.length) {
           return new Response('Bad request: agentId and messages required', { status: 400 })
         }

         // Load agent config (RLS enforces tenant isolation)
         const { data: agent } = await supabase
           .from('agents')
           .select('*')
           .eq('id', agentId)
           .single()

         if (!agent) return new Response('Agent not found', { status: 404 })

         // Update agent status to running
         await supabase.from('agents').update({ status: 'running' }).eq('id', agentId)

         // Initialize providers and get model
         initProviders()
         const adapter = getProvider(agent.model_provider)
         const model = adapter.getModel(agent.model_name)

         try {
           const result = await streamText({
             model,
             system: agent.system_prompt ?? undefined,
             messages,
             temperature: Number(agent.temperature) || 0.7,
             maxTokens: agent.max_tokens ?? 4096,
             onFinish: async () => {
               // Reset agent status to idle on completion
               await supabase.from('agents').update({ status: 'idle' }).eq('id', agentId)
             },
           })

           return result.toDataStreamResponse()
         } catch (error) {
           // Set agent status to error on failure
           await supabase.from('agents').update({ status: 'error' }).eq('id', agentId)
           const message = error instanceof Error ? error.message : 'Stream failed'
           return new Response(JSON.stringify({ error: message }), { status: 500 })
         }
       }
       ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client -> /api/chat route | Untrusted messages, agentId, conversationId cross to streaming endpoint |
| /api/chat -> LLM provider | User messages forwarded to external LLM API |
| LLM response -> Client | Streamed tokens rendered in browser |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-09 | Spoofing | /api/chat route | mitigate | `getUser()` check returns 401 before any DB or LLM calls |
| T-03-10 | Elevation of Privilege | Cross-tenant agent access via agentId | mitigate | Agent fetched via Supabase with RLS; `agents_select` policy enforces `tenant_id = get_current_tenant_id()` |
| T-03-11 | Tampering | Prompt injection via user messages | accept | System prompt passed separately in `streamText({ system })`, not concatenated with user input; further prompt defense is Phase 4+ scope |
| T-03-12 | Denial of Service | Excessive token generation | mitigate | `maxTokens` from agent config (validated by Zod max 128000); per-tenant rate limiting deferred to Phase 6 (BILL-03) |
| T-03-13 | Information Disclosure | API key in response | mitigate | API keys from server env vars only; never included in agent query response or streamed data |
| T-03-14 | Tampering | Code block rendering | accept | React JSX auto-escapes HTML in text content; code blocks rendered inside pre/code elements as safe text nodes, never raw HTML injection |
</threat_model>
