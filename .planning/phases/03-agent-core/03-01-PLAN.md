       import type { AIProviderAdapter } from './adapter'
       import { OpenAIAdapter } from './openai-adapter'
       import { AnthropicAdapter } from './anthropic-adapter'

       const registry = new Map<string, AIProviderAdapter>()

       export function registerProvider(adapter: AIProviderAdapter): void {
         registry.set(adapter.provider, adapter)
       }

       export function getProvider(name: string): AIProviderAdapter {
         const adapter = registry.get(name)
         if (!adapter) throw new Error(`Unknown provider: ${name}`)
         return adapter
       }

       export function initProviders(): void {
         if (registry.size > 0) return
         registerProvider(new OpenAIAdapter())
         registerProvider(new AnthropicAdapter())
       }

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client -> Server Actions | Untrusted form data crosses into server actions |
| Server Action -> Supabase | Server actions query DB; must enforce tenant isolation |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Spoofing | Agent actions | mitigate | Every action calls `getUser()` first; returns error on null user |
| T-03-02 | Elevation of Privilege | Agent create/update/delete | mitigate | `hasPermission()` check with `agent:create/update/delete` before DB operation |
| T-03-03 | Tampering | Agent form data | mitigate | `agentFormSchema.parse(formData)` with Zod before insert; rejects invalid temperature/tokens/provider |
| T-03-04 | Elevation of Privilege | Cross-tenant agent access | mitigate | RLS policies on `agents` table enforce `tenant_id = get_current_tenant_id()`; queries always filter by tenant_id |
| T-03-05 | Information Disclosure | AI provider API keys | mitigate | API keys read from server env vars only; never stored in agents table or returned to client |
</threat_model>
